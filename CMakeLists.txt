cmake_minimum_required(VERSION 3.28)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_CURRENT_SOURCE_DIR}/cmake")
project(CubeChess)
project(CubeChessTest)
add_subdirectory(tests)
add_subdirectory(src)
add_executable(exec main.c)
add_executable(unit_tests tests/test_main.c)
target_link_libraries (CubeChess LINK_PUBLIC sqlite3)
target_link_libraries (CubeChessTest LINK_PUBLIC sqlite3)
target_link_libraries(unit_tests PRIVATE cunit)
target_link_libraries(unit_tests PUBLIC CubeChessTest)
target_link_libraries(exec PUBLIC CubeChess)
